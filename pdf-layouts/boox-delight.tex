% boox-delight.tex — preamble for the boox-delight pandoc layout.
% Loaded via `include-in-header` from boox-delight.yaml.
% Adopts the cool editorial subset of Jelle's UIDESIGNSTYLE.md tasteID palette,
% sized for the Boox Note Air 3C (A5, Kaleido 3 color e-ink).
%
% REQUIRES TeX Live 2021 or newer: colortbl pre-2021 does not patch booktabs
% rules to honor \arrayrulecolor, so the soft-gray table rules silently fall
% back to black on older distros. Compile-clean either way; only visual.

\usepackage{xcolor}
% UIDESIGNSTYLE.md tasteID palette
\definecolor{boxnavy}{HTML}{12436D}      % rgb(18,67,109)  — primary heading
\definecolor{boxteal}{HTML}{28A197}      % rgb(40,161,151) — H2 / link accent
\definecolor{boxburgundy}{HTML}{801650}  % rgb(128,22,80)  — H3 / strong
\definecolor{boxbody}{HTML}{3D3D3D}      % rgb(61,61,61)   — body text
\definecolor{boxrule}{HTML}{BFBFBF}      % rgb(191,191,191) — table rules
\definecolor{boxcodebg}{HTML}{F4EFE7}    % warm light gray — code panel
\definecolor{boxcodefr}{HTML}{D9CFC0}    % warm gray — code panel border

% Body color (soft black on e-ink, easier on the eyes than #000)
\AtBeginDocument{\color{boxbody}}

% 15% smaller body via direct font-size override
\usepackage{anyfontsize}
\renewcommand{\normalsize}{\fontsize{9.5}{12.4}\selectfont}
\AtBeginDocument{\normalsize}

% Editorial heading hierarchy
\usepackage{titlesec}
\titleformat{\section}{\normalfont\Large\bfseries\color{boxnavy}}{\thesection}{0.6em}{}
\titleformat{\subsection}{\normalfont\large\bfseries\color{boxteal}}{\thesubsection}{0.6em}{}
\titleformat{\subsubsection}{\normalfont\normalsize\bfseries\color{boxburgundy}}{\thesubsubsection}{0.6em}{}
\titleformat{\paragraph}[runin]{\normalfont\normalsize\bfseries\color{boxburgundy}}{}{0em}{}[.]
\titlespacing*{\section}{0pt}{1.4ex plus 0.4ex minus 0.2ex}{0.6ex plus 0.2ex}
\titlespacing*{\subsection}{0pt}{1.1ex plus 0.3ex minus 0.2ex}{0.5ex plus 0.2ex}
\titlespacing*{\subsubsection}{0pt}{0.9ex plus 0.2ex minus 0.1ex}{0.4ex plus 0.1ex}

% Code blocks: subtle warm-gray panel via tcolorbox wrapping verbatim / Shaded.
% pandoc emits `Shaded`+`Highlighting` for fences with a language tag, plain
% `verbatim` for untagged fences. We wrap both with the same tcolorbox skin.
%
% TRADE-OFF: hooking `verbatim` globally also wraps any raw `\begin{verbatim}`
% block a markdown author embeds for un-styled literal text. Because this
% layout is opt-in (you invoke it explicitly via `md2pdf.sh foo.md boox-delight`)
% the panel is the design intent — every literal block looks like a code box.
% Documents that need un-styled verbatim should pick a different layout.
\usepackage{tcolorbox}
\tcbuselibrary{breakable,skins}
\usepackage{fancyvrb}
\usepackage{etoolbox}

\newcommand{\boxcodeopen}{%
  \begin{tcolorbox}[breakable,enhanced,colback=boxcodebg,colframe=boxcodefr,
    boxrule=0.4pt,arc=1pt,left=4pt,right=4pt,top=2pt,bottom=2pt,
    before skip=4pt,after skip=4pt,
    fontupper=\footnotesize]%
}
\newcommand{\boxcodeclose}{\end{tcolorbox}}

% Hook plain pandoc-emitted `verbatim` (untagged fences, command blocks).
\BeforeBeginEnvironment{verbatim}{\boxcodeopen}
\AfterEndEnvironment{verbatim}{\boxcodeclose}

% If pandoc loaded its highlighting macros (language-tagged fences), wrap Shaded too.
\AtBeginDocument{%
  \expandafter\ifx\csname Shaded\endcsname\relax\else
    \renewenvironment{Shaded}{\boxcodeopen}{\boxcodeclose}%
  \fi
}

% Inline code — monospace, slightly smaller, no panel
\let\oldtexttt\texttt
\renewcommand{\texttt}[1]{{\fontsize{8.6}{10.8}\selectfont\oldtexttt{#1}}}

% Table rules in soft gray, looser breathing room
\usepackage{array}
\usepackage{tabularx}
\usepackage{booktabs}
\usepackage{colortbl}        % needed for \arrayrulecolor
\arrayrulecolor{boxrule}
\renewcommand{\arraystretch}{1.15}
\setlength{\tabcolsep}{4pt}

% Minimal header / footer — page number bottom center in slate
\usepackage{fancyhdr}
\pagestyle{fancy}
\fancyhf{}
\renewcommand{\headrulewidth}{0pt}
\renewcommand{\footrulewidth}{0pt}
\fancyfoot[C]{\color{boxbody}\small\thepage}
